    table.insert(SortedAreaFarmAreas, {Name = name, ID = id})
end
table.sort(SortedAreaFarmAreas, function(a, b) return a.ID < b.ID end)

-- Reverse lookup: AreaID -> proper display name, built from the same
-- table the area button list uses. Lets detection show "Mobius Circus"
-- instead of the internal folder codename "youleyuan", etc.
AreaIdToName = {}
for name, id in pairs(AreaFarmIDs) do
    AreaIdToName[id] = name
end

-- Automatic area detection via region containment — tested and confirmed
-- working. Checks the player's position against every area's zone
-- BasePart(s) under workspace.Area.<AreaName>.Area.AreaZone, using
-- local-space bounds checking (point-in-OBB test), no button click or
-- live tracking needed.
--
-- IMPORTANT: the folder names here (Root, Ice, center, Caidan, ...) do
-- NOT match the display names in AreaFarmIDs (Starter Island, Frost
-- Isle, Skyheart Isle, ...) — "Volcano" only worked before by
-- coincidence since both places happen to use that exact word. Instead,
-- the zone part's own name embeds the real numeric AreaID directly
-- (e.g. "Area3" for Ice, "Area10000" for center) — these line up with
-- the existing AreaFarmIDs numbers, so we read the ID straight from the
-- part instead of trying to match names. This also means detection now
-- works for areas that aren't in AreaFarmIDs at all yet (e.g. Caidan,
-- youleyuan, longdao, kulouisland, fish).
function detectCurrentArea()
    local r = getRoot()
    if not r then return nil, nil end
    local myPos = r.Position

    local areaFolder = workspace:FindFirstChild("Area")
    if not areaFolder then return nil, nil end

    for _, region in ipairs(areaFolder:GetChildren()) do
        local areaZone = region:FindFirstChild("Area")
        if areaZone then
            local zoneFolder = areaZone:FindFirstChild("AreaZone")
            if zoneFolder then
                for _, part in ipairs(zoneFolder:GetChildren()) do
                    if part:IsA("BasePart") then
                        local ok, localPos = pcall(function()
                            return part.CFrame:PointToObjectSpace(myPos)
                        end)
                        if ok then
                            local half = part.Size / 2
                            local inside = math.abs(localPos.X) <= half.X
                                and math.abs(localPos.Y) <= half.Y
                                and math.abs(localPos.Z) <= half.Z
                            if inside then
                                local areaId = tonumber(part.Name:match("Area(%d+)"))
                                local niceName = areaId and AreaIdToName[areaId] or region.Name
                                return niceName, areaId
                            end
                        end
                    end
                end
            end
        end
    end
    return nil, nil
end

-- Teleports to a saved AreaFarm spot: fires the area-change remote for
-- its AreaID, waits, then snaps the character to the exact saved CFrame.
-- Shared by both Farming Spot TP and the boss loop's "go back" step.
function teleportToAreaFarmSpot(spot, skipSettle, preSnapWait)
    if not spot then return false end
    pcall(function()
        DataPullFunc:InvokeServer("AreaTeleportToRegionChannel", spot.areaId)
    end)
    task.wait(preSnapWait or 2)

    local r = getRoot()
    if r then
        r.CFrame = CFrame.new(spot.x, spot.y, spot.z)
        r.AssemblyLinearVelocity = Vector3.zero
        r.AssemblyAngularVelocity = Vector3.zero

        local function settle()
            task.wait(5)
            local char = LP.Character
            local hum = char and char:FindFirstChildOfClass("Humanoid")
            if hum then
                for i = 1, 2 do
                    hum:ChangeState(Enum.HumanoidStateType.Jumping)
                    task.wait(6)
                end
            end
        end

        if skipSettle then
            task.spawn(settle)
        else
            settle()
        end

        return true
    end
    return false
end

-- Startup Check: Runs ONCE on script load/rejoin. Wait for game & character
-- to fully load before teleporting to saved spot.
task.spawn(function()
    if not game:IsLoaded() then
        pcall(function() game.Loaded:Wait() end)
    end
    task.wait(3.5) -- Extended initial wait on rejoin so remote channels are ready

    if farmingSpot then
        local r = getRoot()
        if not r then
            local startWait = os.clock()
            while os.clock() - startWait < 3 do
                r = getRoot()
                if r then break end
                task.wait(0.2)
            end
        end

        if r then
            local currentAreaName, currentAreaId = detectCurrentArea()
            local savedPos = Vector3.new(farmingSpot.x, farmingSpot.y, farmingSpot.z)
            local dist = (r.Position - savedPos).Magnitude

            if currentAreaId ~= farmingSpot.areaId or dist > 20 then
                FarmSpotStatusLbl.Text = "Startup TP to saved spot..."
                FarmSpotStatusLbl.TextColor3 = Color3.fromRGB(205, 215, 245)
                teleportToAreaFarmSpot(farmingSpot, true, 5)
                FarmSpotStatusLbl.Text = farmSpotStatusText()
                FarmSpotStatusLbl.TextColor3 = C.green
            end
        end
    end
end)

-- ==========================================
-- BOSS TRACKING — fully event-driven. Replaces the old fixed-position
-- bookmark + UTC 15-min timer entirely: no more guessing a schedule and
-- polling "is a boss here yet" -- instead this reacts directly to the
-- server's own spawn notification (NotificationEvent), and travels to
-- wherever the boss actually is.
-- ==========================================

-- Finds the exact monster instance in ClientMonsters whose resolved
-- nameplate matches the boss name from the notification -- no distance
-- cap needed here (unlike the old findNearestBoss), since we already
-- know exactly which area we just teleported into.
function findBossByName(bossName)
    buildNameCache()
    for _, m in ipairs(clientMonsters:GetChildren()) do
        local info = nameCache[m.Name]
        local displayName = info and info.name or m.Name
        if displayName == bossName then
            return m
        end
    end
    return nil
end

-- Waits for ClientMonsters' child count to stop changing (settled) or a
-- max timeout -- avoids scanning before the new area has actually
-- streamed its monsters in.
function waitForMonstersToSettle(maxWait, bossName)
    local start = os.clock()
    local lastCount = -1
    local stableSince = nil
    while os.clock() - start < maxWait do
        if bossName and wasBossDeathNotified(bossName) then
            return false, true -- second return value indicates died/killed
        end
        local count = #clientMonsters:GetChildren()
        if count == lastCount then
            stableSince = stableSince or os.clock()
            if os.clock() - stableSince >= 1 then
                return true, false
            end
        else
            stableSince = nil
        end
        lastCount = count
        task.wait(0.25)
    end
    return false, false
end

_G.NoxarBossDeathNotified = _G.NoxarBossDeathNotified or {}

function markBossDeathNotified(bossName)
    if not bossName then return end
    local deadName = (type(bossName) == "table" and bossName[1]) or tostring(bossName)
    _G.NoxarBossDeathNotified[deadName] = os.clock()

    -- Purge from bossQueue immediately so it won't queue or waste time on it
    local removedCount = 0
    if bossQueue then
        for i = #bossQueue, 1, -1 do
            if bossQueue[i] and bossQueue[i].name == deadName then
                table.remove(bossQueue, i)
                removedCount = removedCount + 1
            end
        end
    end

    -- Live UI status update
    if bossLoopEnabled and BossLoopLiveStatusLbl then
        if currentBossFight and currentBossFight.name == deadName then
            BossLoopLiveStatusLbl.Text = "Status: 💀 " .. deadName .. " died — aborting..."
            BossLoopLiveStatusLbl.TextColor3 = C.red
        elseif removedCount > 0 then
            BossLoopLiveStatusLbl.Text = "Status: 💀 " .. deadName .. " died — removed from queue (" .. #bossQueue .. " left)"
            BossLoopLiveStatusLbl.TextColor3 = C.red
        end
    end
end

function wasBossDeathNotified(bossName, sinceTime)
    local t = bossName and _G.NoxarBossDeathNotified and _G.NoxarBossDeathNotified[bossName]
    return t ~= nil and (not sinceTime or t >= sinceTime)
end

function resnapToCurrentBoss(bossName, fallbackMonster)
    local monster = findBossByName(bossName) or fallbackMonster
    if not monster then return fallbackMonster end
    local root = monster:FindFirstChild("Root") or monster.PrimaryPart
    if root then
        _G.NoxarSetBossMount(true)
        task.wait(0.1)
        glueToTarget(root)
        task.wait(0.1)
        _G.NoxarSetBossMount(false)
    end
    return monster
end

-- Fights a specific boss instance until it dies. Teleports onto it once
-- (same one-shot style as the saved-farm-spot TP button) and then just
-- watches for it to disappear from ClientMonsters -- damage is expected
-- to come from your in-game auto-attack (e.g. equipped pets), not from
-- this script clicking anything. "Still present in ClientMonsters" is
-- the same death check used everywhere else in this file. Bails out
-- early (returns false) if bossLoopEnabled goes false mid-fight.
function fightBossUntilDead(monster, isSpecial, bossName)
    local root = monster:FindFirstChild("Root") or monster.PrimaryPart
    if not root then
        local startRootWait = os.clock()
        while os.clock() - startRootWait < 2 do
            root = monster:FindFirstChild("Root") or monster.PrimaryPart
            if root then break end
            task.wait(0.1)
        end
    end

    local fightStartedAt = os.clock()
    local lastSnapAt = fightStartedAt

    if root then
        BossLoopLiveStatusLbl.Text = "Status: Snapping onto " .. tostring(bossName or "boss") .. "..."
        BossLoopLiveStatusLbl.TextColor3 = C.gold

        _G.NoxarSetBossMount(true)
        task.wait(0.1)
        glueToTarget(root)
        task.wait(0.1)
        _G.NoxarSetBossMount(false)
    end

    BossLoopLiveStatusLbl.Text = "Status: Fighting " .. tostring(bossName or "boss") .. "..."
    BossLoopLiveStatusLbl.TextColor3 = C.gold

    while bossLoopEnabled do
        if wasBossDeathNotified(bossName, fightStartedAt) then
            return true -- server kill notification confirmed it
        end

        if not clientMonsters:FindFirstChild(monster.Name) then
            return true -- boss is dead/disappeared from ClientMonsters
        end

        -- Teleport/glitch failsafe: if we have been "fighting" the same target
        -- for 5s with no MonsterDead notification, snap directly back onto the
        -- live boss model again. This does NOT area-teleport, only direct snap.
        if os.clock() - lastSnapAt >= 5 then
            BossLoopLiveStatusLbl.Text = "Status: Re-snapping to " .. tostring(bossName or "boss") .. "..."
            monster = resnapToCurrentBoss(bossName, monster) or monster
            lastSnapAt = os.clock()
        end

        if isSpecial and os.clock() - fightStartedAt >= 9 then
            BossLoopLiveStatusLbl.Text = "Status: " .. tostring(bossName or "Special") .. " timed out after 9s — continuing..."
            return true -- special/ultra timeout guard so the loop never gets stuck here
        end

        task.wait(0.5) -- just polling for death now, no need to run at ATTACK_DELAY speed
    end

    return false -- got turned off mid-fight
end

-- Runs the full cycle for one queued boss: travel to its area, scan for
-- it by name, teleport onto it, fight until dead.
function runBossFight(entry)
    currentBossFight = entry

    -- Check if this boss was already killed while waiting in the queue
    if wasBossDeathNotified(entry.name) then
        BossLoopLiveStatusLbl.Text = "Status: 💀 " .. entry.name .. " was killed — skipping!"
        BossLoopLiveStatusLbl.TextColor3 = C.red
        task.wait(1)
        currentBossFight = nil
        return
    end

    _G.NoxarSetBossMount(true) -- mount while traveling between boss/special areas
    BossLoopLiveStatusLbl.Text = "Status: " .. entry.name .. " @ " .. entry.area .. " — traveling..."
    BossLoopLiveStatusLbl.TextColor3 = C.gold

    local areaId = AreaFarmIDs[entry.area]
    if not areaId then
        BossLoopLiveStatusLbl.Text = "Status: ❌ Unknown area for " .. entry.name .. " (" .. entry.area .. ")"
        BossLoopLiveStatusLbl.TextColor3 = C.red
        currentBossFight = nil
        return
    end

    pcall(function()
        DataPullFunc:InvokeServer("AreaTeleportToRegionChannel", areaId)
    end)
    task.wait(2)
    if not bossLoopEnabled then currentBossFight = nil; return end

    if wasBossDeathNotified(entry.name) then
        BossLoopLiveStatusLbl.Text = "Status: 💀 " .. entry.name .. " killed during travel — skipping!"
        BossLoopLiveStatusLbl.TextColor3 = C.red
        task.wait(1)
        currentBossFight = nil
        return
    end

    BossLoopLiveStatusLbl.Text = "Status: Waiting for monsters to load in " .. entry.area .. "..."
    local settled, died = waitForMonstersToSettle(6, entry.name)
    if not bossLoopEnabled or died or wasBossDeathNotified(entry.name) then
        if died or wasBossDeathNotified(entry.name) then
            BossLoopLiveStatusLbl.Text = "Status: 💀 " .. entry.name .. " killed while loading — skipping!"
            BossLoopLiveStatusLbl.TextColor3 = C.red
            task.wait(1)
        end
        currentBossFight = nil
        return
    end

    BossLoopLiveStatusLbl.Text = "Status: Searching for " .. entry.name .. "..."
    local monster = nil
    local searchStartedAt = os.clock()
    local maxFindWait = 15

    while bossLoopEnabled and os.clock() - searchStartedAt < maxFindWait do
        if wasBossDeathNotified(entry.name) then
            BossLoopLiveStatusLbl.Text = "Status: 💀 " .. entry.name .. " killed while searching — skipping!"
            BossLoopLiveStatusLbl.TextColor3 = C.red
            task.wait(1)
            currentBossFight = nil
            return
        end

        monster = findBossByName(entry.name)
        if monster then
            break -- found early, do not wait the rest of the timeout
        end

        local remaining = math.max(0, maxFindWait - (os.clock() - searchStartedAt))
        BossLoopLiveStatusLbl.Text = "Status: Searching for " .. entry.name .. "... " .. string.format("%.0f", remaining) .. "s"
        task.wait(0.25)
    end

    if not bossLoopEnabled then currentBossFight = nil; return end

    if wasBossDeathNotified(entry.name) then
        BossLoopLiveStatusLbl.Text = "Status: 💀 " .. entry.name .. " killed — skipping!"
        BossLoopLiveStatusLbl.TextColor3 = C.red
        task.wait(1)
        currentBossFight = nil
        return
    end

    if not monster then
        BossLoopLiveStatusLbl.Text = "Status: ❌ " .. entry.name .. " not found after 15s search"
        BossLoopLiveStatusLbl.TextColor3 = C.red
        currentBossFight = nil
        return
    end

    local isSpecial = isSpecialBossName(entry.name)
    BossLoopLiveStatusLbl.Text = "Status: Fighting " .. entry.name .. "..."
    BossLoopLiveStatusLbl.TextColor3 = C.gold
    local won = fightBossUntilDead(monster, isSpecial, entry.name)

    if won then
        _G.NoxarSetBossMount(true) -- remount after kill so next travel/return is mounted
        BossLoopLiveStatusLbl.Text = "Status: " .. entry.name .. " defeated! Waiting 2s..."
        task.wait(2)
    end

    currentBossFight = nil
end

function isAlreadyTracked(bossName)
    if currentBossFight and currentBossFight.name == bossName then return true end
    for _, q in ipairs(bossQueue) do
        if q.name == bossName then return true end
    end
    return false
end

connectTracked("notificationEvent", NotificationEvent.OnClientEvent, function(...)
    local args = table.pack(...) -- table.pack, not ipairs -- some slots are nil and ipairs would silently stop early

    -- Death notifications use ServerMessage/MonsterDead and include the dead
    -- monster name at args[4][1]. Track them so active fights can end instantly.
    if args[1] == "ServerMessage" and args[2] == "MonsterDead" then
        local payload = args[4]
        if type(payload) == "table" then
            markBossDeathNotified(payload[1])
        end
        return
    end

    if not bossLoopEnabled then return end
    if args[1] ~= "ViewBubble" then return end

    -- Regular bosses use "MonsterAppear"; Ultra Rare monsters use
    -- "UltraRareMonsterAppear". Both payload shapes include Name and
    -- DescParams[2] = area name, so the same queue/fight path can handle both.
    local eventKind = args[2]
    if eventKind ~= "MonsterAppear" and eventKind ~= "UltraRareMonsterAppear" then return end

    local payload = args[4]
    if type(payload) ~= "table" then return end

    local bossName = payload.Name
    local areaName = payload.DescParams and payload.DescParams[2]
    if not bossName or not areaName then return end

    -- Clear stale death notification on a fresh spawn so future fights can queue properly
    if _G.NoxarBossDeathNotified then
        _G.NoxarBossDeathNotified[bossName] = nil
    end

    if not selectedBosses[bossName] then return end -- not one you've opted in to
    if isAlreadyTracked(bossName) then return end

    table.insert(bossQueue, { name = bossName, area = areaName })
    if not currentBossFight then
        BossLoopLiveStatusLbl.Text = "Status: " .. bossName .. " spawned @ " .. areaName .. " — queued (" .. #bossQueue .. ")"
    else
        BossLoopLiveStatusLbl.Text = "Status: Fighting " .. currentBossFight.name .. " — " .. bossName .. " queued next (" .. #bossQueue .. ")"
    end
end)

-- Dispatcher: whenever nothing's currently being fought and the queue has
-- something waiting, pop the next one (FIFO) and run it -- never
-- interrupting whatever fight is already in progress. When the queue
-- drains completely, return to the farming spot and cleanly resume the
-- regular Auto-Farm loop if it was on before the boss loop took over.
spawnTracked("bossLoop", function()
    local returnedToFarmSpot = true -- starts true so it doesn't try to "return" before ever leaving
    while task.wait(0.5) do
        if not bossLoopEnabled then
            returnedToFarmSpot = true
            continue
        end

        if not currentBossFight and #bossQueue > 0 then
            returnedToFarmSpot = false
            if running then
                wasFarmingBeforeBossFight = true
                toggleFarming(false) -- cleanly stop Auto-Farm (updates its button too), not just flipping `running`
            else
                wasFarmingBeforeBossFight = false
            end

            local nextEntry = table.remove(bossQueue, 1)
            runBossFight(nextEntry)

        elseif not currentBossFight and #bossQueue == 0 and not returnedToFarmSpot then
            if farmingSpot then
                BossLoopLiveStatusLbl.Text = "Status: Queue empty — returning to Farming Spot..."
                BossLoopLiveStatusLbl.TextColor3 = Color3.fromRGB(205, 215, 245)
                -- Teleport/travel back to saved Farming Spot (stays mounted during travel)
                teleportToAreaFarmSpot(farmingSpot, true, 5)
            end
            _G.NoxarSetBossMount(false, true) -- Unmount AFTER travel/teleport back to Farming Spot completes
            if wasFarmingBeforeBossFight then
                toggleFarming(true) -- cleanly restart Auto-Farm (spawns a fresh farmLoop thread, updates its button)
                wasFarmingBeforeBossFight = false
            end
            BossLoopLiveStatusLbl.Text = "Status: Waiting for a boss spawn..."
            BossLoopLiveStatusLbl.TextColor3 = Color3.fromRGB(205, 215, 245)
            returnedToFarmSpot = true
        end
    end
end)
