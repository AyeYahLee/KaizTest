-- ==========================================
-- 0.5 BACKGROUND THREAD REGISTRY
-- ==========================================
-- Every infinite task.spawn loop in this script is registered here so it
-- can be cancelled in one place. This fixes two problems:
--   1) Re-running the script (common while testing) used to stack a brand
--      new copy of every loop on top of the old ones, since nothing ever
--      cancelled the previous instance's threads.
--   2) The panel's Close (X) button used to just destroy the GUI without
--      stopping any of these loops, so they kept polling/firing forever
--      in the background even with no UI left to see them.
-- _G.NoxarPanelThreads persists across script re-executions (unlike a
-- local table), so on load we cancel whatever the previous run left
-- running, then start fresh.
if _G.NoxarPanelThreads then
    for _, thread in pairs(_G.NoxarPanelThreads) do
        if type(task.cancel) == "function" then pcall(task.cancel, thread) end
    end
end
_G.NoxarPanelThreads = {}

-- spawnTracked(name, fn): same as task.spawn(fn), but stores the thread
-- under `name` in the registry so it can be looked up and cancelled later
-- (by CloseBtn, or by the next script re-execution above).
function spawnTracked(name, fn)
    if _G.NoxarPanelThreads[name] then
        if type(task.cancel) == "function" then pcall(task.cancel, _G.NoxarPanelThreads[name]) end
    end
    local thread = task.spawn(fn)
    _G.NoxarPanelThreads[name] = thread
    return thread
end

function cancelAllTrackedThreads()
    for _, thread in pairs(_G.NoxarPanelThreads) do
        if type(task.cancel) == "function" then pcall(task.cancel, thread) end
    end
    _G.NoxarPanelThreads = {}
end

-- _G.NoxarPanelConnections is the same idea as _G.NoxarPanelThreads above,
-- but for event connections instead of threads. This matters because a
-- few listeners here connect to persistent objects that outlive the
-- script (NotificationEvent is a RemoteEvent, promptOverlay is a CoreGui
-- object, RunService.Heartbeat is a service) -- a plain local connection
-- variable can't be reached by a fresh script re-execution, so without
-- this registry every re-run stacked a brand new listener on top of any
-- still-alive one from the previous run.
if _G.NoxarPanelConnections then
    for _, conn in pairs(_G.NoxarPanelConnections) do
        pcall(function() conn:Disconnect() end)
    end
end
_G.NoxarPanelConnections = {}

-- connectTracked(name, signal, fn): same as signal:Connect(fn), but
-- stores the connection under `name` in the registry so it can be
-- disconnected later (by CloseBtn, or by the next script re-execution
-- above).
function connectTracked(name, signal, fn)
    if _G.NoxarPanelConnections[name] then
        pcall(function() _G.NoxarPanelConnections[name]:Disconnect() end)
    end
    local conn = signal:Connect(fn)
    _G.NoxarPanelConnections[name] = conn
    return conn
end

function disconnectAllTrackedConnections()
    for _, conn in pairs(_G.NoxarPanelConnections) do
        pcall(function() conn:Disconnect() end)
    end
    _G.NoxarPanelConnections = {}
end

-- ==========================================
-- 1.6 PET BREEDING — BAG SCANNING
-- ==========================================
-- Uses the game's own GamePlayer.pet:IterBagItem() iterator (found via
-- decompiled PetBagView) rather than guessing at inventory structure.
-- Filters exactly the same way the in-game breed-parent picker does:
-- not a child pet, Grade >= ParentGradeMin, template not CantBreed,
-- and split by gender (Male -> father list, Female -> mother list).
function getGamePlayer()
    local ok, gp = pcall(function()
        local CPM = require(RS.CommonLibrary.Player.ClientPlayerManager)
        return CPM.GetGamePlayer()
    end)
    if ok then return gp end
    return nil
end

-- Returns two arrays: { {id=.., name=.., grade=..}, ... } for males and females
function scanBreedablePets()
    local males, females = {}, {}
    local gp = getGamePlayer()
    if not gp or not gp.pet or not gp.pet.IterBagItem then
        return males, females
    end

    local ParentGradeMin = CfgPetBreed and CfgPetBreed.ParentGradeMin or nil

    local MaleType, FemaleType = PetGenderType.Male, PetGenderType.Female

    local ok = pcall(function()
        gp.pet:IterBagItem(function(item)
            if item:IsChildPet() then return end
            local tmpl = item:GetTmpl()
            if tmpl and tmpl.CantBreed then return end
            local grade = item:GetGrade()
            if ParentGradeMin and grade < ParentGradeMin then return end

            local entry = {
                id = item:GetId(),
                name = (tmpl and tmpl.Name) or ("Pet #" .. tostring(item:GetId())),
                grade = grade,
            }

            local gender = item:GetGender()
            if MaleType ~= nil and gender == MaleType then
                table.insert(males, entry)
            elseif FemaleType ~= nil and gender == FemaleType then
                table.insert(females, entry)
            end
        end)
    end)

    return males, females
end

end

-- unlockTarget: disconnects the persistent farm-lock connection (see
-- lockOntoTarget below). Tracked via connectTracked, so a stale lock
-- from a previous script re-execution also gets cleaned up automatically
-- at the top of the file, same as every other tracked connection here.
function unlockTarget()
    if _G.NoxarPanelConnections["farmLock"] then
        pcall(function() _G.NoxarPanelConnections["farmLock"]:Disconnect() end)
        _G.NoxarPanelConnections["farmLock"] = nil
    end
end

-- lockOntoTarget: this is the original continuous glue behavior --
-- a RunService.Heartbeat connection that re-locks the player's CFrame
-- onto the target every single frame, which is what makes regular
-- Auto-Farm feel smooth/instant instead of snapping in discrete steps.
-- Used only by the regular farm loop (startFarming), which calls this
-- once per acquired target and unlockTarget() when it's done with it --
-- NOT called repeatedly per attack tick, since the Heartbeat connection
-- itself is what keeps it locked every frame in between.
function lockOntoTarget(targetRoot)
    unlockTarget()
    if not targetRoot or not targetRoot.Parent then return end

    local initialPos = targetRoot.Position
    local _, monsterYRotation, _ = targetRoot.CFrame:ToOrientation()

    groundRayParams.FilterDescendantsInstances = {
        LP.Character,
        targetRoot.Parent,
        clientMonsters
    }

    local rayOrigin = Vector3.new(initialPos.X, initialPos.Y + 5, initialPos.Z)
    local raycastResult = workspace:Raycast(rayOrigin, rayDirection, groundRayParams)
    local groundY = raycastResult and raycastResult.Position.Y or (initialPos.Y - 4)

    local lockedCFrame = CFrame.new(initialPos.X, groundY + HOVER_HEIGHT, initialPos.Z) * CFrame.fromOrientation(0, monsterYRotation, 0)

    connectTracked("farmLock", RunService.Heartbeat, function()
        local r = getRoot()
        if r and targetRoot and targetRoot.Parent then
            r.CFrame = lockedCFrame
            r.AssemblyLinearVelocity = Vector3.zero
            r.AssemblyAngularVelocity = Vector3.zero
        else
            unlockTarget()
        end
    end)
end

-- glueToTarget: teleports the player on top of the target once (a
-- "snap TP"), instead of holding a persistent connection. This is used
-- only by the boss fight (fightBossUntilDead), which snaps once when
-- the fight starts and then relies on your in-game auto-attack -- it
-- doesn't need continuous re-locking the way regular farming does.
function glueToTarget(targetRoot)
    if not targetRoot or not targetRoot.Parent then return end
    local r = getRoot()
    if not r then return end

    local pos = targetRoot.Position
    local _, monsterYRotation, _ = targetRoot.CFrame:ToOrientation()

    groundRayParams.FilterDescendantsInstances = {
        LP.Character,
        targetRoot.Parent,
        clientMonsters
    }

    local rayOrigin = Vector3.new(pos.X, pos.Y + 5, pos.Z)
    local raycastResult = workspace:Raycast(rayOrigin, rayDirection, groundRayParams)
    local groundY = raycastResult and raycastResult.Position.Y or (pos.Y - 4)

    r.CFrame = CFrame.new(pos.X, groundY + HOVER_HEIGHT, pos.Z) * CFrame.fromOrientation(0, monsterYRotation, 0)
    r.AssemblyLinearVelocity = Vector3.zero
    r.AssemblyAngularVelocity = Vector3.zero
end

function buildNameCache()
    table.clear(nameCache) 
    if not clientMonsters then return end

    local selfGui = workspace:FindFirstChild("SelfGUI")
    if not selfGui then return end

    for _, gui in ipairs(selfGui:GetChildren()) do
        if not gui:IsA("BillboardGui") then continue end
        local adornee = gui.Adornee
        if not adornee then continue end
        local candidates = { adornee.Parent and adornee.Parent.Parent, adornee.Parent }
        for _, monster in ipairs(candidates) do
            if monster and monster.Parent == clientMonsters then
                local labName = gui:FindFirstChild("LabName", true)
                local labLv = gui:FindFirstChild("LabLv", true)
                if labName and labName:IsA("TextLabel") and labName.Text ~= "" then
                    nameCache[monster.Name] = { name = labName.Text, level = labLv and labLv.Text or "" }
                end
                break
            end
        end
    end
end

function scanMonsterTypes()
    buildNameCache()
    local seen = {}
    local types = {}

    for _, m in ipairs(clientMonsters:GetChildren()) do
        local info = nameCache[m.Name]
        local displayName = info and info.name or m.Name
        local root = m:FindFirstChild("Root")
        local pos = root and root:IsA("BasePart") and root.Position or (m.PrimaryPart and m.PrimaryPart.Position)
        
        if not seen[displayName] then
            seen[displayName] = true
            table.insert(types, {displayName = displayName, level = info and info.level or "", count = 1, pos = pos})
        else
            for _, t in ipairs(types) do
                if t.displayName == displayName then t.count += 1; break end
            end
        end
    end
    table.sort(types, function(a,b) return a.displayName < b.displayName end)
    return types
end

function getTargetMonsters()
    local monsters = {}
    for _, m in ipairs(clientMonsters:GetChildren()) do
        local info = nameCache[m.Name]
        local displayName = info and info.name or m.Name
        if selectedMonster and not displayName:lower():find(selectedMonster:lower(), 1, true) then
            continue
        end
        local root = m:FindFirstChild("Root")
        local pos = root and root:IsA("BasePart") and root.Position or (m.PrimaryPart and m.PrimaryPart.Position)
        if pos then table.insert(monsters, {model = m, pos = pos, root = root}) end
    end
    return monsters
end

function startFarming(statusLbl)
    local myId = runId
    local killed = 0

    while running and runId == myId do
        buildNameCache()
        local monsters = getTargetMonsters()

        if #monsters == 0 then
            currentTarget = nil
            unlockTarget()
            unequipAll()
            statusLbl.Text = "Status: Waiting for spawns..."
            task.wait(0.5)
            continue
        end

        local target = monsters[1]
        currentTarget = target.model

        if target.root then
            lockOntoTarget(target.root)
        end
        
        unequipAll()
        task.wait(EQUIP_DELAY)
        
        if not running or runId ~= myId then break end
        
        if clientMonsters:FindFirstChild(target.model.Name) then
            equipBest()
        else
            continue
        end

        while running and runId == myId do
            if not clientMonsters:FindFirstChild(target.model.Name) then
                killed += 1
                unlockTarget()
                unequipAll()
                currentTarget = nil
                statusLbl.Text = "Status: Killed " .. killed .. " (" .. (selectedMonster or "All") .. ")"
                break
            end
            
            local root = target.model:FindFirstChild("Root")
            local cd = root and root:FindFirstChildOfClass("ClickDetector")
            if cd then pcall(function() fireclickdetector(cd) end) end
            
            task.wait(ATTACK_DELAY)
        end
    end
    
    currentTarget = nil
    unlockTarget()
    unequipAll()
    statusLbl.Text = "Status: Stopped. Killed: " .. killed
end

ClaimStatus = nil
LastClaimLbl = nil
function claimRemote(channelName, ...)
    local args = table.pack(...)
    pcall(function()
        DataPullFunc:InvokeServer(channelName, table.unpack(args, 1, args.n))
    end)
    task.wait(0.1)
end

function claimEverything()
    if not autoClaimEnabled then
        return false
    end

    if ClaimStatus then
        ClaimStatus.Text = "Status: Claiming..."
    end


    -- Daily tasks
    for _, id in ipairs({101, 102, 103, 104, 105}) do
        claimRemote("TaskClaimRewardChannel", id)
    end

    -- Weekly tasks
    for _, id in ipairs({201, 202, 203, 204, 205}) do
        claimRemote("TaskClaimRewardChannel", id)
    end

    -- Tower rewards
    claimRemote("TowerTakeDailyRewardAllChannel")

    -- Daily chest/reward claims
    claimRemote("ChestRewardTakeDailyChestChannel", 1)
    claimRemote("ChestRewardTakeDailyRewardChannel", 1)

    if ClaimStatus then
        ClaimStatus.Text = "Status: Done!"
    end
    if LastClaimLbl then
        LastClaimLbl.Text = "Last claim: " .. os.date("%H:%M:%S")
    end

    return true
end

function rejoinSameServer()
    local placeId = game.PlaceId
    local ok, joinData = pcall(function() return LP:GetJoinData() end)
    local code = ok and joinData and joinData.PrivateServerAccessCode

    if code and code ~= "" then
        task.wait(3) 
        pcall(function()
            TeleportService:TeleportToPrivateServer(placeId, code, {LP})
        end)
    else
        task.wait(2)
        pcall(function() TeleportService:Teleport(placeId, LP) end)
    end
end
