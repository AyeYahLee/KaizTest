CloseBtn.MouseButton1Click:Connect(function()
    running = false
    runId += 1
    unequipAll()

    -- The boss loop reuses the same equip/attack helpers as regular
    -- farming and its glueToTarget call is just a one-shot teleport, so
    -- there's no separate connection to worry about there. The regular
    -- farm loop's persistent lock (lockOntoTarget/"farmLock") gets torn
    -- down by disconnectAllTrackedConnections() below along with
    -- everything else in the registry -- just clear the boss queue state
    -- so nothing tries to resume a fight after the panel is gone.
    bossLoopEnabled = false
    table.clear(bossQueue)
    currentBossFight = nil
    _G.NoxarSetBossMount(false, true)

    cancelAllTrackedThreads()
    disconnectAllTrackedConnections()

    -- Clear globals that hold direct runtime references. Helper functions can
    -- stay cached, but instance refs / mount closure state should not keep old
    -- objects or old script closures alive after closing.
    _G.NoxarUIScaleControls = nil
    _G.NoxarBossMountState = nil
    _G.NoxarSetBossMount = nil
    if _G.NoxarUI then
        _G.NoxarUI.MainScale = nil
    end

    sg:Destroy()
end)

-- Dragging Engine
dragging, dragStart, startPos = nil, nil, nil
connectTracked("dragInputBegan", TitleBar.InputBegan, function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true; dragStart = input.Position; startPos = Main.Position
    end
end)
connectTracked("dragInputChanged", UIS.InputChanged, function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - dragStart
        Main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)
connectTracked("dragInputEnded", UIS.InputEnded, function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = false
    end
end)

task.spawn(function()
    local types
    for attempt = 1, 8 do
        task.wait(0.3)
        types = scanMonsterTypes()
        local hasRealNames = false
        for _, t in ipairs(types) do
            if not t.displayName:match("^Monster_%d+$") then
                hasRealNames = true
                break
            end
        end
        if hasRealNames or #types == 0 then break end
    end
    populateList(types)
    ScanBtn.Text = "🔍 Rescan (" .. #types .. " Found)"
end)

-- Background Auto Hatch Executor
-- getIncubatorState reads on-screen GUI text, which doesn't reliably
-- update/stream when far from the incubators. Since ClientHatchStart /
-- ClientHatchTaken have no actual distance restriction (confirmed working
-- from another island), we track state ourselves instead of reading the
-- GUI.
--
-- Optimization: we don't know whether resending ClientHatchStart to a
-- slot that's already hatching wastes an egg, and hatch durations are
-- random so we can't just wait "long enough" to be safe. Instead we only
-- ever send Start right after a Claim succeeds on that same slot — i.e.
-- we only start a slot we just emptied ourselves, never blind-restart a
-- slot we're merely guessing is empty. incubatorKnownEmpty tracks which
-- slots we're confident are actually empty right now.
incubatorKnownEmpty = {} -- [index] = true once we've either claimed it or confirmed it empty

-- One-time pass: try Start on every slot once at startup, in case some
-- are sitting empty from before the script ran (e.g. after a rejoin).
hatchBootstrapped = false

spawnTracked("autoHatchLoop", function()
    while task.wait(3) do
        if not autoHatchEnabled or not selectedEggId then continue end

        -- First pass ever: seed every slot as a Start candidate so
        -- slots left empty from before we started still get filled.
        if not hatchBootstrapped then
            hatchBootstrapped = true
            for i = 1, 5 do incubatorKnownEmpty[i] = true end
        end

        for incubatorIndex = 1, 5 do
            if not autoHatchEnabled then break end

            -- Always attempt to claim — if it's not ready, this
            -- should just fail/no-op server-side rather than error.
            -- If it succeeds, the slot is now empty, so mark it for
            -- a Start attempt below.
            local claimedOk = pcall(function()
                if EggSystem then
                    EggSystem.ClientHatchTaken(incubatorIndex)
                else
                    DataPullFunc:InvokeServer("EggHatchTakenChannel", incubatorIndex)
                end
            end)
            if claimedOk then
                incubatorKnownEmpty[incubatorIndex] = true
            end
            task.wait(0.2)

            -- Only Start a slot we know is actually empty (either
            -- just claimed above, or seeded on the first pass) —
            -- never blind-restart a slot that might already be
            -- hatching.
            if incubatorKnownEmpty[incubatorIndex] then
                pcall(function()
                    if EggSystem then
                        EggSystem.ClientHatchStart(incubatorIndex, selectedEggId)
                    else
                        DataPullFunc:InvokeServer("EggHatchStartChannel", incubatorIndex, selectedEggId)
                    end
                end)
                -- Once started, we no longer know it's empty — clear
                -- the flag so we don't Start it again until we've
                -- claimed it fresh.
                incubatorKnownEmpty[incubatorIndex] = nil
                task.wait(0.2)
            end
        end
    end
end)

-- ==========================================
-- 10. BREEDING STATUS WATCHER + AUTO-COLLECT
-- ==========================================
-- Polls GetBreeding(1) the same way AreaPetBreedShower does, updates the
-- status label with a live countdown, and (if enabled) auto-fires
-- ClientHatching() the moment the timer completes. If Auto Breed Loop
-- is also on, immediately re-fires ClientCreateBreeding() with the same
-- two selected pets right after collecting, so the whole
-- breed -> wait -> hatch -> breed cycle repeats with no input needed.
spawnTracked("breedingWatcher", function()
    while task.wait(2) do
        local info = getBreedingSlotInfo()

        if info == nil then
            -- Slot is empty. If Auto Breed is on and we have both pets
            -- selected, start a new breed right away — this covers both
            -- "just collected" and "rejoined with slot already empty".
            if autoBreedEnabled and breedFatherId and breedMotherId and PetBreedSystem then
                BreedStatusLbl.Text = "Slot 1: Empty — auto-breeding again..."
                BreedStatusLbl.TextColor3 = Color3.fromRGB(200, 210, 240)
                local ok = pcall(function()
                    PetBreedSystem.ClientCreateBreeding(breedFatherId, breedMotherId)
                end)
                if ok then
                    BreedStatusLbl.Text = "Auto-breeding: " .. tostring(breedFatherLabel) .. " x " .. tostring(breedMotherLabel)
                    BreedStatusLbl.TextColor3 = C.green
                end
            else
                BreedStatusLbl.Text = "Slot 1: Empty — pick two pets and breed"
                BreedStatusLbl.TextColor3 = Color3.fromRGB(200, 210, 240)
            end
        else
            local serverTime = nil
            pcall(function()
                local CPM = require(RS.CommonLibrary.Player.ClientPlayerManager)
                serverTime = CPM.GetServerTime()
            end)
            serverTime = serverTime or os.time()

            local remaining = (info.StartTick + BreedTimeSeconds) - serverTime

            if remaining > 0 then
                local h = math.floor(remaining / 3600)
                local m = math.floor((remaining % 3600) / 60)
                local s = math.floor(remaining % 60)
                BreedStatusLbl.Text = string.format("Slot 1: Breeding... %02d:%02d:%02d left", h, m, s)
                BreedStatusLbl.TextColor3 = Color3.fromRGB(200, 210, 240)
            else
                BreedStatusLbl.Text = "Slot 1: Egg ready to hatch!"
                BreedStatusLbl.TextColor3 = C.green

                if autoBreedHatchEnabled and PetBreedSystem then
                    local ok = pcall(function()
                        PetBreedSystem.ClientHatching()
                    end)
                    if ok then
                        if autoBreedEnabled and breedFatherId and breedMotherId then
                            -- Collected — immediately queue the next breed.
                            -- The slot won't show empty until next poll,
                            -- so fire it here rather than waiting a full
                            -- extra 2s cycle for the "info == nil" branch.
                            task.wait(0.3)
                            local ok2 = pcall(function()
                                PetBreedSystem.ClientCreateBreeding(breedFatherId, breedMotherId)
                            end)
                            if ok2 then
                                BreedStatusLbl.Text = "Collected! Auto-breeding again: " .. tostring(breedFatherLabel) .. " x " .. tostring(breedMotherLabel)
                                BreedStatusLbl.TextColor3 = C.green
                            end
                        else
                            BreedStatusLbl.Text = "Slot 1: Collected! Pick pets to breed again."
                            BreedStatusLbl.TextColor3 = C.green
                        end
                    end
                end
            end
        end
    end
end)
