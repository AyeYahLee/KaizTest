eggButtons = {} -- {btn = TextButton, searchName = "lowercase egg name"}

function refreshEggListCanvas()
    task.defer(function()
        local currentScale = tonumber(uiScaleValue) or 1
        if currentScale <= 0 then currentScale = 1 end
        local visibleCount = 0
        for _, entry in ipairs(eggButtons) do
            if entry.btn and entry.btn.Visible then
                visibleCount = visibleCount + 1
            end
        end
        local unscaledAbsHeight = (ScrollHatchList.AbsoluteContentSize.Y / currentScale) + 10
        local unscaledCountHeight = visibleCount * 37 + 10
        ScrollHatch.CanvasSize = UDim2.new(0, 0, 0, math.max(unscaledAbsHeight, unscaledCountHeight))
    end)
end

function applyEggFilter(query)
    query = query:lower():match("^%s*(.-)%s*$") -- trim whitespace
    for _, entry in ipairs(eggButtons) do
        entry.btn.Visible = (query == "") or entry.searchName:find(query, 1, true) ~= nil
    end
    refreshEggListCanvas()
end

EggSearchBox:GetPropertyChangedSignal("Text"):Connect(function()
    applyEggFilter(EggSearchBox.Text)
end)

function _G.NoxarUI.breedButtonText(kind, open, label, id)
    -- Use plain ASCII arrows because some Roblox mobile fonts show triangle
    -- glyphs as boxes. Gender symbols stay intact.
    local arrow = open and "^" or "v"
    if kind == "father" then
        if id then
            return "♂ Father " .. arrow .. "  " .. tostring(label) .. " (ID " .. tostring(id) .. ")"
        end
        return "♂ Father " .. arrow .. "  Select male..."
    else
        if id then
            return "♀ Mother " .. arrow .. "  " .. tostring(label) .. " (ID " .. tostring(id) .. ")"
        end
        return "♀ Mother " .. arrow .. "  Select female..."
    end
end

breedListOpenFor = nil -- "father" or "mother" or nil

function closeBreedList()
    -- Important: collapse the list BEFORE restoring LayoutOrder.
    -- If we move BreedListFrame back under Mother while it is still visible,
    -- the father list flashes under the Mother selector for a split second.
    breedListOpenFor = nil
    FatherBtn.Text = _G.NoxarUI.breedButtonText("father", false, breedFatherLabel, breedFatherId)
    MotherBtn.Text = _G.NoxarUI.breedButtonText("mother", false, breedMotherLabel, breedMotherId)
    _G.NoxarUI.animateDrop(BreedListFrame, false, 0)

    task.delay(0.18, function()
        -- Only restore the normal order if another selector was not opened
        -- while the close animation was running.
        if breedListOpenFor == nil then
            MotherBtn.LayoutOrder = 11
            BreedListFrame.LayoutOrder = 12
        end
    end)
end

-- Visual-only grade label for the breeding selection list. Does not change
-- any grade values or breeding logic. Colors intentionally not used.
function getBreedRankVisual(grade)
    if grade == 1 then return "E" end
    if grade == 2 then return "D" end
    if grade == 3 then return "C" end
    if grade == 4 then return "B" end
    if grade == 5 then return "A" end
    if grade == 6 then return "S" end
    if grade == 7 then return "SS" end
    return tostring(grade or "?")
end

function populateBreedList(kind)
    for _, c in ipairs(BreedListFrame:GetChildren()) do
        if c:IsA("TextButton") then c:Destroy() end
    end

    local males, females = scanBreedablePets()
    local list = (kind == "father") and males or females

    if #list == 0 then
        _G.NoxarUI.choiceRow(
            BreedListFrame,
            "No eligible " .. (kind == "father" and "male" or "female") .. " pets found",
            "empty",
            false
        )
    else
        for _, entry in ipairs(list) do
            local rankName = getBreedRankVisual(entry.grade)
            local row = _G.NoxarUI.choiceRow(BreedListFrame, entry.name, rankName, false)
            row.MouseButton1Click:Connect(function()
                if kind == "father" then
                    breedFatherId = entry.id
                    breedFatherLabel = entry.name
                    FatherBtn.Text = _G.NoxarUI.breedButtonText("father", false, entry.name, entry.id)
                    FatherBtn.TextColor3 = C.text
                else
                    breedMotherId = entry.id
                    breedMotherLabel = entry.name
                    MotherBtn.Text = _G.NoxarUI.breedButtonText("mother", false, entry.name, entry.id)
                    MotherBtn.TextColor3 = C.text
                end
                closeBreedList()
                saveConfig()
            end)
        end
    end

    -- Animated dropdown, matching the other selector lists.
    local rowCount = math.max(#list, 1)
    local shownRows = math.min(rowCount, 5)
    local targetHeight = shownRows * 38
    BreedListFrame.CanvasSize = UDim2.new(0, 0, 0, rowCount * 38 + 6)
    _G.NoxarUI.animateDrop(BreedListFrame, true, targetHeight)
end

FatherBtn.MouseButton1Click:Connect(function()
    if breedListOpenFor == "father" then
        closeBreedList()
    else
        breedListOpenFor = "father"
        -- Put the dropdown directly under the father selector, then push the
        -- mother selector down. This fixes the list appearing under Mother.
        BreedListFrame.LayoutOrder = 11
        MotherBtn.LayoutOrder = 12
        FatherBtn.Text = _G.NoxarUI.breedButtonText("father", true, breedFatherLabel, breedFatherId)
        MotherBtn.Text = _G.NoxarUI.breedButtonText("mother", false, breedMotherLabel, breedMotherId)
        populateBreedList("father")
    end
end)

        MotherBtn.Text = _G.NoxarUI.breedButtonText("mother", true, breedMotherLabel, breedMotherId)

        PetBreedSystem.ClientCreateBreeding(breedFatherId, breedMotherId)

    if ok then
        BreedStatusLbl.Text = "Breeding started: " .. tostring(breedFatherLabel) .. " x " .. tostring(breedMotherLabel)

    else
        BreedStatusLbl.Text = "⚠ Breed request failed: " .. tostring(err)
        BreedStatusLbl.TextColor3 = C.red
    end
end)

    if not autoBreedEnabled and (not breedFatherId or not breedMotherId) then
        BreedStatusLbl.Text = "⚠ Select both a father and mother before enabling Auto Breed"
        BreedStatusLbl.TextColor3 = C.red
        return
    end
    autoBreedEnabled = not autoBreedEnabled
    if autoBreedEnabled then
        AutoBreedBtn.BackgroundColor3 = C.green
        AutoBreedBtn.Text = "● AUTO BREED LOOP: ON"
        -- Auto Breed requires Auto-Collect to also be on, or the loop
        -- would breed once and then just sit there forever with a
        -- finished egg nobody collects.
        if not autoBreedHatchEnabled then
            autoBreedHatchEnabled = true
            AutoBreedHatchBtn.BackgroundColor3 = C.green
            AutoBreedHatchBtn.Text = "● AUTO-COLLECT BRED EGG: ON"
        end
    else
        AutoBreedBtn.BackgroundColor3 = C.red
        AutoBreedBtn.Text = "● AUTO BREED LOOP: OFF"
    end
    saveConfig()
end)
