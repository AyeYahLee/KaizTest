_G.NoxarUI = _G.NoxarUI or {}
function _G.NoxarUI.corner(obj, r)
    local c = obj:FindFirstChildOfClass("UICorner") or Instance.new("UICorner", obj)
    c.CornerRadius = UDim.new(0, r or 8)
    return c
end
function _G.NoxarUI.stroke(obj, color, trans, thick)
    local s = obj:FindFirstChildOfClass("UIStroke") or Instance.new("UIStroke", obj)
    s.Color = color or C.border
    s.Transparency = trans or 0.4
    s.Thickness = thick or 1
    return s
end
function _G.NoxarUI.grad(obj, a, b, rot)
    local g = obj:FindFirstChildOfClass("UIGradient") or Instance.new("UIGradient", obj)
    g.Color = ColorSequence.new(a or C.surface, b or C.card)
    g.Rotation = rot or 90
    return g
end
function _G.NoxarUI.nav(btn, active)
    btn.BorderSizePixel = 0
    btn.AutoButtonColor = false
    btn.BackgroundTransparency = 0
    btn.BackgroundColor3 = active and Color3.fromRGB(33, 38, 58) or Color3.fromRGB(14, 17, 27)
    btn.TextColor3 = active and C.text or C.sub
    btn.Font = Enum.Font.GothamSemibold
    btn.TextSize = 11
    btn.TextXAlignment = Enum.TextXAlignment.Left
    _G.NoxarUI.corner(btn, 8)
    _G.NoxarUI.stroke(btn, active and C.accent or C.borderSoft, active and 0.25 or 0.75, 1)
    if not btn:FindFirstChildOfClass("UIPadding") then
        Instance.new("UIPadding", btn).PaddingLeft = UDim.new(0, 12)
    end
end
function _G.NoxarUI.drop(btn)
    btn.BorderSizePixel = 0
    btn.AutoButtonColor = true
    btn.BackgroundColor3 = C.card2
    btn.TextColor3 = C.text
    btn.Font = Enum.Font.GothamSemibold
    btn.TextSize = 11
    btn.TextXAlignment = Enum.TextXAlignment.Left
    _G.NoxarUI.corner(btn, 8)
    _G.NoxarUI.stroke(btn, C.borderSoft, 0.45, 1)
    if not btn:FindFirstChildOfClass("UIPadding") then
        Instance.new("UIPadding", btn).PaddingLeft = UDim.new(0, 10)
    end
end

function _G.NoxarUI.animateDrop(frame, open, height)
    height = height or 100
    if open then
        frame.Visible = true
        frame.ClipsDescendants = true
        frame.Size = UDim2.new(1, -20, 0, 0)
        TS:Create(frame, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Size = UDim2.new(1, -20, 0, height)}):Play()
    else
        frame.ClipsDescendants = true
        local tw = TS:Create(frame, TweenInfo.new(0.16, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {Size = UDim2.new(1, -20, 0, 0)})
        tw:Play()
        task.delay(0.17, function()
            if frame and frame.Parent and frame.Size.Y.Offset <= 2 then
                frame.Visible = false
            end
        end)
    end
end

function _G.NoxarUI.updateChoiceRow(row, pill, bar, active, activeText, inactiveText)
    if not row then return end
    row.BackgroundColor3 = active and Color3.fromRGB(36, 44, 68) or Color3.fromRGB(17, 21, 32)
    local stroke = row:FindFirstChildOfClass("UIStroke")
    if stroke then
        stroke.Color = active and C.accent or C.borderSoft
        stroke.Transparency = active and 0.25 or 0.72
    end
    if bar then
        bar.BackgroundColor3 = active and C.accent2 or C.borderSoft
    end
    if pill then
        pill.BackgroundColor3 = active and C.accent or C.card2
        pill.TextColor3 = active and C.text or C.sub
        pill.Text = active and (activeText or "Selected") or (inactiveText or "Select")
    end
end

-- Manual UI scale. Auto-resize was removed because it made the panel jump on
-- some devices/minimize states. The Settings tab controls this value and it is
-- saved in NoxarSave.json.
_G.NoxarUI.MainScale = Instance.new("UIScale", Main)
function _G.NoxarUI.applyUIScale(centerMode)
    uiScaleValue = math.clamp(tonumber(uiScaleValue) or 1, 0.75, 1.20)

    if centerMode == "keep" and Main and Main.Parent then
        -- Preserve the panel's CURRENT visual center while changing scale.
        -- This makes + / - resizing grow/shrink from the middle instead of
        -- looking like it slides sideways from the top-left.
        local oldCenter = Main.AbsolutePosition + (Main.AbsoluteSize / 2)
        _G.NoxarUI.MainScale.Scale = uiScaleValue
        task.defer(function()
            if Main and Main.Parent then
                local newSize = Main.AbsoluteSize
                Main.Position = UDim2.fromOffset(
                    math.floor(oldCenter.X - newSize.X / 2),
                    math.floor(oldCenter.Y - newSize.Y / 2)
                )
            end
        end)
    else
        -- Initial load/rejoin: center the scaled panel at the usual starting
        -- spot using the full open panel size.
        _G.NoxarUI.MainScale.Scale = uiScaleValue
        Main.Position = UDim2.new(
            0.5,
            -math.floor((normalSize.X.Offset * uiScaleValue) / 2),
            0.42,
            -math.floor((normalSize.Y.Offset * uiScaleValue) / 2)
        )
    end
end
function _G.NoxarUI.setManualScale(value)
    uiScaleValue = math.clamp(tonumber(value) or uiScaleValue or 1, 0.75, 1.20)
    _G.NoxarUI.applyUIScale("keep")
    if _G.NoxarUIScaleControls and _G.NoxarUIScaleControls.Box then
        _G.NoxarUIScaleControls.Box.Text = string.format("%.2f", uiScaleValue)
    end
    saveConfig()
end
_G.NoxarUI.applyUIScale()
