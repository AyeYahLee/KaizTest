function populateList(types)
    for _, c in ipairs(ScrollFarm:GetChildren()) do
        if c:IsA("TextButton") then c:Destroy() end
    end

    local allBtn = _G.NoxarUI.choiceRow(ScrollFarm, "Everything", "ALL", selectedMonster == nil)
    allBtn.MouseButton1Click:Connect(function()
        selectedMonster = nil
        TargetLbl.Text = "Target: All Monsters"
        TargetLbl.TextColor3 = C.green
        TargetDropBtn.Text = "▼  All Monsters"
        _G.NoxarUI.animateDrop(ScrollFarm, false, 132)
        saveConfig()
    end)

    for _, t in ipairs(types) do
        local lvlStr = (t.level ~= "") and t.level or ""
        local meta = "x" .. tostring(t.count)
        if lvlStr ~= "" then meta = lvlStr .. " • " .. meta end
        local btn = _G.NoxarUI.choiceRow(ScrollFarm, t.displayName, meta, selectedMonster == t.displayName)
        btn.MouseButton1Click:Connect(function()
            selectedMonster = t.displayName
            TargetLbl.Text = "Target: " .. t.displayName
            TargetLbl.TextColor3 = C.gold
            TargetDropBtn.Text = "▼  " .. t.displayName
            _G.NoxarUI.animateDrop(ScrollFarm, false, 132)
            saveConfig()
        end)
    end
    ScrollFarm.CanvasSize = UDim2.new(0, 0, 0, ScrollList.AbsoluteContentSize.Y + 8)
end

ScanBtn.MouseButton1Click:Connect(function()

            types = scanMonsterTypes()

            for _, t in ipairs(types) do

function toggleFarming(forceState)
    if forceState ~= nil then
        running = forceState
    else
        running = not running
    end
    
    saveConfig()

    if running then
        runId += 1
        StartFarmBtn.BackgroundColor3 = C.red
        StartFarmBtn.TextColor3 = C.text
        StartFarmBtn.Text = "⏹ Halt Farming"
        spawnTracked("farmLoop", function() startFarming(FarmStatus) end)
    else
        StartFarmBtn.BackgroundColor3 = C.green
        StartFarmBtn.TextColor3 = C.bg
        StartFarmBtn.Text = "▶ Start Auto Farm"
        unlockTarget()
    end
end

function getPathToolSafe()
    if PathTool then return PathTool end
    if _G.PathTool then
        PathTool = _G.PathTool
        return PathTool
    end
    pcall(function()
        PathTool = require(game:GetService("ReplicatedStorage").CommonLibrary.Tool.PathTool)
    end)
    return PathTool
end

function openViewSafe(viewName, arg1)
    -- Fully guarded: no nil function calls, no startup require, no console spam
    -- unless the game's own opened view prints something internally.
    pcall(function()
        local pt = getPathToolSafe()
        local vm = pt and pt.ViewManager
        local openView = vm and vm.OpenView
        if type(openView) ~= "function" then return end
        if arg1 ~= nil then
            openView(viewName, arg1)
        else
            openView(viewName)
        end
    end)
end

function setRejoin(state)
    rejoinTimerEnabled = state
    if rejoinTimerEnabled then
        rejoinDeadline = tick() + rejoinMinutes * 60
        RejoinToggleBtn.BackgroundColor3 = C.green
        RejoinToggleBtn.TextColor3 = C.bg
        RejoinToggleBtn.Text = "ON"
    else
        rejoinDeadline = nil
        RejoinToggleBtn.BackgroundColor3 = C.red
        RejoinToggleBtn.TextColor3 = C.text
        RejoinToggleBtn.Text = "OFF"
        RejoinStatusLbl.Text = "Disabled"
    end
    saveConfig()
end

RejoinToggleBtn.MouseButton1Click:Connect(function()
    setRejoin(not rejoinTimerEnabled)
end)

RejoinBox.FocusLost:Connect(function()
    rejoinMinutes = tonumber(RejoinBox.Text) or 60
    RejoinBox.Text = tostring(rejoinMinutes)
    if rejoinTimerEnabled then
        rejoinDeadline = tick() + rejoinMinutes * 60
    end
    saveConfig()
end)

spawnTracked("rejoinTimer", function()
    while true do
        task.wait(1)
        if rejoinTimerEnabled and rejoinDeadline then
            local remaining = rejoinDeadline - tick()
            if remaining <= 0 then
                RejoinStatusLbl.Text = "Rejoining now..."
                rejoinSameServer()
            else
                local m = math.floor(remaining / 60)
                local s = math.floor(remaining % 60)
                RejoinStatusLbl.Text = string.format("Rejoin in: %02d:%02d", m, s)
            end
        end
    end
end)

function refreshPositionsList()
    for _, c in ipairs(ScrollPos:GetChildren()) do
        if c:IsA("Frame") then c:Destroy() end
    end
    
    for i, posData in ipairs(savedPositions) do
        local row = Instance.new("Frame", ScrollPos)
        row.Size = UDim2.new(1, -10, 0, 26)
        row.BackgroundColor3 = C.card
        Instance.new("UICorner", row).CornerRadius = UDim.new(0, 7)
        
        local nameBox = Instance.new("TextBox", row)
        nameBox.Size = UDim2.new(1, -65, 1, 0)
        nameBox.Position = UDim2.new(0, 6, 0, 0)
        nameBox.BackgroundTransparency = 1
        nameBox.TextColor3 = C.text
        nameBox.Font = Enum.Font.Gotham
        nameBox.TextSize = 10
        nameBox.Text = posData.name
        nameBox.TextXAlignment = Enum.TextXAlignment.Left
        nameBox.ClearTextOnFocus = false
        
        nameBox.FocusLost:Connect(function()
            if nameBox.Text == "" then nameBox.Text = posData.name end
            posData.name = nameBox.Text
            saveConfig()
        end)
        
        local tpBtn = Instance.new("TextButton", row)
        tpBtn.Size = UDim2.new(0, 30, 0, 20)
        tpBtn.Position = UDim2.new(1, -58, 0.5, -10)
        tpBtn.BackgroundColor3 = C.accent
        tpBtn.TextColor3 = C.text
        tpBtn.Font = Enum.Font.GothamBold
        tpBtn.TextSize = 10
        tpBtn.Text = "TP"
        Instance.new("UICorner", tpBtn).CornerRadius = UDim.new(0, 7)
        
        tpBtn.MouseButton1Click:Connect(function()
            local r = getRoot()
            if r then
                r.CFrame = CFrame.new(posData.x, posData.y, posData.z)
            end
        end)
        
        local delBtn = Instance.new("TextButton", row)
        delBtn.Size = UDim2.new(0, 20, 0, 20)
        delBtn.Position = UDim2.new(1, -24, 0.5, -10)
        delBtn.BackgroundColor3 = Color3.fromRGB(150, 40, 40)
        delBtn.TextColor3 = C.text
        delBtn.Font = Enum.Font.GothamBold
        delBtn.TextSize = 10
        delBtn.Text = "X"
        Instance.new("UICorner", delBtn).CornerRadius = UDim.new(0, 7)
        
        delBtn.MouseButton1Click:Connect(function()
            table.remove(savedPositions, i)
            saveConfig()
            refreshPositionsList() 
        end)
    end

end
